// ENHANCED VERSION WITH PRIVACY CONTROLS function doPost(e) { try { // Parse the JSON data from the form const data = JSON.parse(e.postData.contents); // Get or create the spreadsheet (PRIVATE - only you can access) const sheet = getSpreadsheet(); // Add headers if this is the first submission if (sheet.getLastRow() === 0) { sheet.getRange(1, 1, 1, 11).setValues([[ 'Timestamp', 'School Name', 'School Type', 'Contact Name', 'Contact Role', 'Email', 'Phone', 'Events', 'Student Count', 'Grade Levels', 'Receive Updates' ]]); // Format header row sheet.getRange(1, 1, 1, 11) .setBackground('#4361ee') .setFontColor('white') .setFontWeight('bold'); } // Add the new row with form data const newRow = sheet.getLastRow() + 1; sheet.getRange(newRow, 1, 1, 11).setValues([[ data.timestamp, data.schoolName, data.schoolType, data.contactName, data.contactRole, data.email, data.phone, data.events, data.students, data.gradeLevels, data.updates ? 'Yes' : 'No' ]]); // Format the new row sheet.getRange(newRow, 1, 1, 11) .setBorder(true, true, true, true, true, true) .setVerticalAlignment('middle'); // Auto-resize columns for better readability sheet.autoResizeColumns(1, 11); // Send email notifications (ONLY TO YOU) sendAdminNotification(data); // Send confirmation to school (ONLY TO THEM) sendConfirmationEmail(data); // Return success response return ContentService .createTextOutput(JSON.stringify({ result: 'success', message: 'Registration received successfully!' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { console.error('Error processing form:', error); return ContentService .createTextOutput(JSON.stringify({ result: 'error', message: 'We encountered an issue, but your registration was saved. We will contact you shortly.' })) .setMimeType(ContentService.MimeType.JSON); } } function getSpreadsheet() { // PRIVATE SPREADSHEET - only accessible by you const spreadsheetName = 'ViblySpace Event Registrations - PRIVATE'; let spreadsheet; try { const files = DriveApp.getFilesByName(spreadsheetName); if (files.hasNext()) { spreadsheet = SpreadsheetApp.open(files.next()); } else { spreadsheet = SpreadsheetApp.create(spreadsheetName); // MAKE SPREADSHEET PRIVATE (default is already private to you) const file = DriveApp.getFileById(spreadsheet.getId()); file.setSharing(DriveApp.Access.PRIVATE, DriveApp.Permission.VIEW); } } catch (error) { spreadsheet = SpreadsheetApp.create(spreadsheetName); } return spreadsheet.getSheets()[0]; } function sendAdminNotification(data) { try { // ONLY SEND TO ADMIN EMAIL (Update this to your email) const adminEmails = [ 'your-admin-email@gmail.com', // Primary admin // 'second-admin@viblyspace.org', // Optional: add team members ]; const subject = `🎉 New Event Registration: ${data.schoolName}`; const htmlBody = `

🔒 New School Registration

ViblySpace Events Program - ADMIN NOTIFICATION

🔐 CONFIDENTIAL: This information is private and should only be accessed by authorized ViblySpace team members.

📋 Registration Details

🏫 School: ${data.schoolName}

🎓 Type: ${data.schoolType}

👤 Contact: ${data.contactName} (${data.contactRole})

📧 Email: ${data.email}

📞 Phone: ${data.phone}

📅 Events: ${data.events}

👥 Students: ${data.students}

📚 Grades: ${data.gradeLevels}

🔔 Updates: ${data.updates ? 'Yes' : 'No'}

⏰ Submitted: ${new Date(data.timestamp).toLocaleString()}

Next Steps:

  • Contact the school within 2 business days
  • Send event details and schedule
  • Add to your outreach tracking
📊 Data Location: This information is stored in a private Google Sheet accessible only to you.

This email was automatically generated by ViblySpace Events System

🔒 Private and Confidential

`; // Send to all admin emails adminEmails.forEach(email => { if (email && email !== 'your-admin-email@gmail.com') { MailApp.sendEmail({ to: email, subject: subject, htmlBody: htmlBody, body: `New registration from ${data.schoolName}. Check the private spreadsheet for details.` }); } }); } catch (error) { console.error('Error sending admin notification:', error); } }
