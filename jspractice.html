  • Increment the count

// Example: Counter logic
let count = 0;
let step = 1;

function incrementCounter() {
  count += step;
  updateCounterDisplay();
}

function decrementCounter() {
  count -= step;
  updateCounterDisplay();
}

function resetCounter() {
  count = 0;
