Advanced Lists & List Methods

Building on Grade 7 Knowledge

Remember lists from Grade 7? Now we'll learn powerful methods to work with lists more efficiently. Lists are like smart containers that can grow, shrink, and reorganize themselves.

Essential List Methods

1. Adding Items to Lists

# Create a list of fruits
fruits = ["apple", "banana"]

# append() - add to the end
fruits.append("orange")
print(fruits) # ['apple', 'banana', 'orange']

# insert() - add at specific position
fruits.insert(1, "grape")
print(fruits) # ['apple', 'grape', 'banana', 'orange']

# extend() - add multiple items
fruits.extend(["mango", "pineapple"])
print(fruits)
['apple', 'banana', 'orange']
['apple', 'grape', 'banana', 'orange']
['apple', 'grape', 'banana', 'orange', 'mango', 'pineapple']

2. Removing Items from Lists

numbers = [10, 20, 30, 40, 30]

# remove() - remove first matching value
numbers.remove(30)
print(numbers) # [10, 20, 40, 30]

# pop() - remove by index and return value
removed = numbers.pop(1)
print(f"Removed: {removed}") # Removed: 20
print(numbers) # [10, 40, 30]

# clear() - remove all items
numbers.clear()
print(numbers) # []

List Sorting and Searching

# Sorting lists
grades = [85, 92, 78, 96, 88]

# sort() - sort the list itself
grades.sort()
print("Sorted grades:", grades)

# sorted() - return sorted copy
reverse_grades = sorted(grades, reverse=True)
print("Reverse order:", reverse_grades)

# Finding items
print("Index of 92:", grades.index(92))
print("Count of 88:", grades.count(88))

Dictionaries - Key-Value Pairs

What are Dictionaries?

Dictionaries store data as key-value pairs, like a real dictionary where you look up a word (key) to find its meaning (value). They're perfect for storing related information together.

Real-World Example

Think of a student record: you have a name, grade, subjects, and marks. A dictionary lets you store all this information together in one organized place!

Creating and Using Dictionaries

# Creating a dictionary
student = {
    "name": "Anya Sharma",
    "grade": 8,
    "subjects": ["Math", "Science", "English"],
    "marks": {"Math": 85, "Science": 92, "English": 78}
}

# Accessing values
print("Student:", student["name"])
print("Grade:", student["grade"])
print("Math marks:", student["marks"]["Math"])

# Adding new information
student["school"] = "Delhi Public School"
student["attendance"] = 95.5

# Updating values
student["grade"] = 9 # Promoted to next grade!

Dictionary Methods

# Useful dictionary methods
student = {"name": "Rohan", "grade": 8, "city": "Mumbai"}

# get() - safe way to access values
print("Age:", student.get("age", "Not specified"))

# keys() - get all keys
print("Keys:", list(student.keys()))

# values() - get all values
print("Values:", list(student.values()))

# items() - get key-value pairs
print("Items:")
for key, value in student.items():
    print(f" {key}: {value}")

Tuples & Sets

Understanding Tuples

Tuples are like lists but immutable - once created, they cannot be changed. They're perfect for storing data that shouldn't be modified, like coordinates or configuration settings.

Working with Tuples

# Creating and using tuples
# Student information that shouldn't change
student_info = ("Anya Sharma", 14, "Grade 8", "DPS")

# Accessing tuple elements
print("Name:", student_info[0])
print("Age:", student_info[1])
print("Grade:", student_info[2])

# Tuple unpacking
name, age, grade, school = student_info
print(f"Unpacked: {name} is {age} years old in {grade}")

# This would cause an error (tuples are immutable):
# student_info[1] = 15 # Uncomment to see the error

Understanding Sets

Sets are collections of unique items - no duplicates allowed! They're great for mathematical operations and removing duplicates.

# Working with sets
# Create sets of favorite subjects
anya_subjects = {"Math", "Science", "English", "Art"}
rohan_subjects = {"Science", "History", "Math", "Music"}

# Set operations
print("Common subjects (intersection):", anya_subjects & rohan_subjects)
print("All subjects (union):", anya_subjects | rohan_subjects)
print("Subjects only Anya likes:", anya_subjects - rohan_subjects)

# Adding and removing from sets
anya_subjects.add("Computer Science")
anya_subjects.remove("Art")
print("Updated Anya's subjects:", anya_subjects)

String Manipulation

Advanced String Operations

Strings are more than just text - they're sequences of characters that we can manipulate in powerful ways. Let's explore advanced string methods!

Essential String Methods

# Common string operations
text = " Hello, Grade 8 Programmers! "

# Basic string methods
print("Original:", text)
print("Strip spaces:", text.strip())
print("Uppercase:", text.upper())
print("Lowercase:", text.lower())
print("Title case:", text.strip().title())

# String searching
print("Contains 'Grade':", "Grade" in text)
print("Starts with 'Hello':", text.strip().startswith("Hello"))
print("Index of 'Programmers':", text.find("Programmers"))

# String splitting and joining
words = text.strip().split(" ")
print("Split words:", words)
print("Joined with dashes:", "-".join(words))

String Formatting Techniques

# Different ways to format strings
name = "Anya"
age = 14
grade = 8.5

# f-strings (modern and recommended)
message1 = f"Hello {name}, you are {age} years old in grade {grade}!"
print(message1)

# format() method
message2 = "Hello {}, you are {} years old in grade {}!".format(name, age, grade)
print(message2)

# With number formatting
message3 = f"Grade average: {grade:.2f}" # 2 decimal places
print(message3)

# Multi-line f-string
report = f"""
Student Report:
Name: {name}
Age: {age}
Grade: {grade:.1f}
Status: {'Pass' if grade >= 5 else 'Fail'}
"""
print(report)

Final Project: Student Management System

Create a complete system to manage multiple students using dictionaries and lists.

# Student Management System - Complete Project
students = []

def add_student():
    print("\n➕ Add New Student")
    name = input("Enter student name: ")
    grade = int(input("Enter grade: "))
    city = input("Enter city: ")

    student = {
        "name": name,
        "grade": grade,
        "city": city,
        "student_id": len(students) + 1001
    }

    students.append(student)
    print(f"✅ Student {name} added successfully! ID: {student['student_id']}")

def display_students():
    if not students:
        print("📝 No students in the system.")
        return

    print("\n📊 Student List:")
    print("-" * 40)
    for student in students:
        print(f"ID: {student['student_id']}")
        print(f"Name: {student['name']}")
        print(f"Grade: {student['grade']}")
        print(f"City: {student['city']}")
        print("-" * 40)

def find_student():
    name = input("Enter student name to search: ")
    for student in students:
        if student["name"].lower() == name.lower():
            print(f"✅ Found: {student}")
            return
    print("❌ Student not found.")

# Main program loop
while True:
    print("\n🎓 STUDENT MANAGEMENT SYSTEM")
    print("1. Add Student")
    print("2. Display All Students")
    print("3. Find Student")
    print("4. Exit")

    choice = input("Enter your choice (1-4): ")

    if choice == "1":
        add_student()
    elif choice == "2":
        display_students()
    elif choice == "3":
        find_student()
    elif choice == "4":
        print("👋 Thank you for using Student Management System!")
        break
    else:
        print("❌ Invalid choice. Please try again.")
